12/// Initialized values in a scope.
13#[derive(Clone)]
14pub struct SymbolTable {
15    values: HashMap<String, SymbolInfo>,
16}
17
18/// Types of symbol values in a symbol table.
19#[derive(Clone, Debug, PartialEq, Eq, Default)]
20pub enum SymbolValue {
21    /// No expression
22    #[default]
23    NoAssociatedValue,
24
25    /// An enum's value (variants)
26    EnumValue { 
27        /// Variants of the enum
28        variants: Vec<String>,
29    },
30
31    /// A struct's value (fields)
32    StructValue { 
33        /// Fields of the struct
34        fields: Vec<(String, DataType)>,
35    },
36    
37    /// A function's value (params, return type)
38    FunctionValue { 
39        /// Parameters of the function
40        parameters: Vec<(String, DataType)>, 
41
42    },
43}

45/// Information on a symbol in a symboltable.
46#[derive(Clone, Debug)]
47pub struct SymbolInfo {
48    data_type: DataType,
49    value: SymbolValue,
50}
51
52impl SymbolInfo {
53    /// Creates a new `SymbolInfo` instance with the given input paraneters.
54    ///
55    /// This initializer sets up a `SymbolInfo` to keep track of information about
56    /// a particular symbol in a `SymbolTable` using the input data type and 
57    /// symbol value. 
58    ///
59    /// # Parameters
60    ///
61    /// - `data_type`: A `DataType` representing the data type of the symbol.
62    /// - `value`: A `SymbolValue` representing the symbol's value.
63    ///
64    /// # Returns
65    ///
66    /// Returns a new `SymbolInfo` instance incorporating the input parameters.
67    pub fn new(data_type: DataType, value: SymbolValue) -> Self {
68        Self {
69            data_type,
70            value,
71        }
72    }
73
74    /// Returns a clone of this `SymbolInfo`'s value
75    ///
76    /// # Returns
77    ///
78    /// - `SymbolValue` - The `SymbolInfo`'s value as a cloned value.
79    pub fn get_value(&self) -> SymbolValue {
80        self.value.clone()
81    }
82
83    /// Returns a clone of this `SymbolInfo`'s data type
84    ///
85    /// # Returns
86    ///
87    /// - `DataType` - The `SymbolInfo`'s data type as a cloned value.
88    pub fn get_data_type(&self) -> DataType {
89        self.data_type.clone()
90    }
91}

94impl SymbolTable {
95    /// Creates a new `SymbolTable` instance.
96    ///
97    /// This initializer sets up a `SymbolTable` to keep track of individual tables of 
98    /// symbols inside a symbol table stack. 
99    ///
100    /// # Returns
101    ///
102    /// Returns a new `SymbolTable` instance with an empty table.
103    pub fn new() -> Self {
104        SymbolTable {
105            values: HashMap::new(),
106        }
107    }
108
109    /// Adds a new symbol and its `SymbolInfo` to the table.
110    ///
111    /// # Parameters
112    ///
113    /// - `name`: The `String` name of the symbol being added.
114    /// - `info`: A `SymbolInfo` instance representing the info associated with this symbol.
115    ///
116    pub fn add(&mut self, name: String, info: SymbolInfo) {
117        self.values.insert(name, info);
118    }
119
120    /// Retrieves a value from the symbol table.
121    /// 
122    /// # Parameters
123    ///
124    /// - `name`: The `String` name of the symbol being retrieved.
125    ///
126    /// # Returns
127    ///
128    /// - `Option<&SymbolInfo>` - The `SymbolInfo` of a symbol if it exists, else None.
129    ///
130    pub fn get(&self, name: &str) -> Option<&SymbolInfo> {
131        self.values.get(name)
132    }
133
134    /// Checks if the symbol table is empty
135    ///
136    /// # Returns
137    ///
138    /// - `bool` - true if this table is empty, false otherwise.
139    ///
140    pub fn is_empty(&self) -> bool {
141        self.values.is_empty()
142    }
143}
144
145impl fmt::Debug for SymbolTable {
146    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
147        let mut builder = f.debug_map();
148        for (key, value) in &self.values {
149            builder.entry(&key, &value);
150        }
151        builder.finish()
152    }
153}

155/// A stack of symbol tables, used to represent different levels of scopes for an AST's symbols.
156#[derive(Clone)]
157pub struct SymbolTableStack {
158    elements: Vec<SymbolTable>,
159}
160
161impl SymbolTableStack {
162    /// Drives the symbol table stack generation process.
163    ///
164    /// # Parameters
165    ///
166    /// - `ast`: An `AST` instance to generate the symbol table stack for.
167    ///
168    /// # Returns
169    ///
170    /// Returns a `Result` containing an `AST` and `SymbolTableStack` pair if Ok and a vector
171    /// of errors if any errors were encountered.
172    pub fn gen_sym_table_stack(ast: AST) -> Result<(AST, SymbolTableStack), Vec<ErrorType>> {
173        let _ = ast;
174        unimplemented!()
175    }
176
177    /// Routes the generation of the SymbolTableStack based on the type of node encountered.
178    /// 
179    /// # Parameters
180    ///
181    /// - `node`: A reference to an `ASTNode` to use in generating the next iteration of the symbol table stack.
182    /// 
183    /// # Returns
184    ///
185    /// Returns a `Result<(), Vec<ErrorType>>` with Ok indicating the process was successful or
186    /// Err containing a vector of errors encountered if any were encountered.
187    ///
188    /// # Errors
189    ///
190    /// Returns a vector of errors if there was a problem during STS generation.
191    pub fn sym_table_stack_router(&mut self, node: &ASTNode) -> Result<(), Vec<ErrorType>> {
192        let _ = node;
193        unimplemented!()
194    }
195
196
197    /// Creates a new `SymbolTableStack` instance.
198    ///
199    /// This initializer sets up a `SymbolTableStack` with no tables in it
200    /// yet. 
201    ///
202    /// # Returns
203    ///
204    /// Returns a new empty `SymbolTableStack` instance.
205    pub fn new() -> Self {
206        SymbolTableStack {
207            elements: Vec::new(),
208        }
209    }
210
211    /// Pushes a new table onto the stack.
212    /// 
213    /// # Parameters
214    ///
215    /// - `item`: A `SymbolTable` to be pushed onto the stack.
216    ///
217    pub fn push(&mut self, item: SymbolTable) {
218        self.elements.push(item);
219    }
220
221    /// Retrieves the size of the stack.
222    /// 
223    /// # Returns
224    ///
225    /// Returns a `usize` of how many `SymbolTable`s exist in the stack.
226    ///
227    pub fn size(&self) -> usize {
228        self.elements.len()
229    }
230
231    /// Retreives a reference to all tables in the stack.
232    /// 
233    /// # Returns
234    ///
235    /// Returns an immutable reference to the current stack.
236    ///
237    pub fn get_elements(&self) -> &Vec<SymbolTable> {
238        &self.elements
239    }
240
241    /// Adds an element to the symbol table at the specified index (or global scope by default).
242    /// 
243    /// # Parameters
244    ///
245    /// - `name`: The `String` name of the symbol to be added.
246    /// - `info`: The `SymbolInfo` to be attached to the name in the tables.
247    /// 
248    /// # Returns
249    ///
250    /// Returns a `Result<(), Vec<ErrorType>>` with Ok indicating the process was successful or
251    /// Err containing an error encountered if any were encountered.
252    pub fn add_element(&mut self, name: String, info: SymbolInfo) -> Result<(), ErrorType> {
253        let index = self.size() - 1;
254        if let Some(table) = self.elements.get_mut(index) {
255            table.add(name, info);
256            Ok(())
257        } else {
258            Err(ErrorType::DevError { message: "Scope index out of range".to_string() })
259        }
260    }
261
262    /// Returns the `SymbolTable` in the symbol table stack at a given index.
263    /// 
264    /// # Parameters
265    ///
266    /// - `index`: The `usize` index of the table in the stack to retrieve.
267    /// 
268    /// # Returns
269    ///
270    /// Returns a `Result<SymbolTable, Vec<ErrorType>>` with Ok containing a clone of the desired 
271    /// `SymbolTable` if the process was successful or Err containing an error encountered if 
272    /// any were encountered.
273    pub fn get_element(&self, index: usize) -> Result<SymbolTable, ErrorType> {
274        if index < self.elements.len() {
275            match self.elements.get(index) {
276                Some(sts) => {
277                    return Ok(sts.clone());
278                },
279                None => {
280                    return Err(ErrorType::DevError{ message: "No element found".to_string()});
281                }
282            }
283        } 
284        panic!("Invalid index: {:} for size {:}", index, self.elements.len())
285    }
286}

306impl PartialEq for SymbolTableStack {
307    fn eq(&self, other: &Self) -> bool {
308        if self.elements.len() != other.elements.len() {
309            return false;
310        }
311
312        for (self_table, other_table) in self.elements.iter().zip(&other.elements) {
313
314            let self_keys: HashSet<_> = self_table.values.keys().collect();
315            let other_keys: HashSet<_> = other_table.values.keys().collect();
316
317            if self_keys != other_keys {
318                return false;
319            }
320
321            for key in self_keys {
322                let current_symbol_info: &SymbolInfo = self_table.get(key).expect("No symbol info found in comparison of tables!");
323                let current_other_symbol_info: &SymbolInfo = other_table.get(key).expect("No symbol info found in comparison of tables!");
324
325                if current_symbol_info.get_value() != current_other_symbol_info.get_value() {
326                    return false;
327                }
328
329                if current_symbol_info.get_data_type() != current_other_symbol_info.get_data_type() {
330                    return false;
331                }
332
333            }
334        }
335
336        true
337    }
338}
