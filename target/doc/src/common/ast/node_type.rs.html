6/// Defines acceptable syntax elements as part of an `AST`.
7///
8/// Each element A different kind of syntactic construct that can appear in source code, such as constants, identifiers, 
9/// operators, and control structures. These elements are used to build a tree representation of the code's syntactic structure.
10#[derive(Debug, Clone, PartialEq, Default, Eq, Hash)]
11pub enum NodeType {
12    /// A lack of expression, often used as a placeholder.
13    #[default]
14    NoExpression,
15
16    /// A literal value, stored as a string.
17    Literal(String),
18
19    /// An identifier, such as a variable name, stored as a string.
20    Identifier(String),
21
22    /// An operator, such as '+', '-', etc., stored as a string.
23    Operator(String),
24
25    /// A data type, linking directly to the `DataType` enum.
26    Type(DataType),
27
28    /// A constant value, stored as a string.
29    Constant(String),
30
31    /// An `if` statement.
32    IfStatement,
33
34    /// An `else` statement.
35    ElseStatement,
36
37    /// A `for` loop.
38    ForLoop,
39
40    /// A `while` loop.
41    WhileLoop,
42
43    /// A `do-while` loop.
44    DoWhileLoop,
45
46    /// A `break` statement.
47    Break,
48
49    /// A `continue` statement.
50    Continue,
51
52    /// A `return` statement.
53    Return,
54
55    /// A `switch` statement.
56    SwitchStatement,
57
58    /// A case in a `switch` statement.
59    Case,
60
61    /// The default branch in a `switch` statement.
62    Default,
63
64    /// An assignment of a value to an existing variable.
65    Assignment,
66
67    /// The initialization of a variable.
68    Initialization,
69
70    /// A function declaration.
71    FunctionDeclaration,
72
73    /// A struct declaration.
74    StructDeclaration,
75
76    /// An enum declaration.
77    EnumDeclaration,
78
79    /// A module-level expression, often used for scoping.
80    ModuleExpression,
81
82    /// A top-level expression in a module or file.
83    TopLevelExpression,
84
85    /// A block of expressions or statements.
86    BlockExpression,
87
88    /// A condition in control flows.
89    Condition,
90
91    /// An action to be taken, typically in control structures.
92    Action,
93
94    /// A variant in enums or other similar structures.
95    Variant,
96
97    /// A value assigned in statements like initialization and assignment.
98    AssignedValue,
99
100    /// A field in a struct or similar data structure.
101    Field,
102
103    /// A parameter in function declarations.
104    Parameter,
105
106    /// A variable, used in various expressions and statements.
107    Variable,
108
109    /// A binary expression, involving two operands and an operator.
110    BinaryExpression,
111
112    /// A unary expression, involving one operand and an operator.
113    UnaryExpression,
114
115    /// A function call.
116    FunctionCall,
117
118    /// An operand in an expression.
119    Operand,
120
121    /// The initializer section of a loop.
122    LoopInitializer,
123
124    /// The increment section of a loop.
125    LoopIncrement,
126}
127
128/// Provides a display implementation for `DataType`.
129///
130/// # Parameters
131/// * `f` - The formatter.
132///
133/// # Returns
134/// * `fmt::Result` - The result of the formatting operation.
135impl fmt::Display for NodeType {
136    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
137        match self {
138            NodeType::Literal(value) => write!(f, "Literal({})", value),
139            NodeType::Identifier(id) => write!(f, "Identifier({})", id),
140            NodeType::Operator(op) => write!(f, "Operator({})", op),
141            NodeType::Type(data_type) => write!(f, "Type({})", data_type),
142            NodeType::Constant(value) => write!(f, "Constant({})", value),
143            _ => write!(f, "{:?}", self) 
144        }
145    }
146}
