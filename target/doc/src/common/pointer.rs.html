1//! Provides utilities for safe, synchronized access to raw LLVM pointers.
2//!
3//! This module defines structures and enums to safely handle raw pointers across multiple threads,
4//! encapsulating them within `Arc` and `RwLock` for concurrent access.
5
6extern crate llvm_sys as llvm;
7use llvm::{execution_engine::LLVMExecutionEngineRef, prelude::{LLVMBasicBlockRef, LLVMBuilderRef, LLVMContextRef, LLVMModuleRef, LLVMTypeRef, LLVMValueRef}};
8use std::{ffi::c_void, ptr::NonNull, sync::{Arc, RwLock}};

10/// Enum to represent various LLVM references for type management.
11#[derive(Debug, Clone, Copy)]
12pub enum LLVMRef {
13    /// Encapsulates an LLVM context, a mechanism for holding the global
14    /// data used throughout the LLVM infrastructure during the compilation process.
15    Context(LLVMContextRef), // https://llvm.org/doxygen/classllvm_1_1LLVMContext.html
16    /// Represents an LLVM module, which is a collection of related code that
17    /// together forms a program or part of a program. It is the primary unit of code packaging in LLVM.
18    Module(LLVMModuleRef), // https://llvm.org/doxygen/classllvm_1_1Module.html
19    /// Represents a value in the LLVM system, which could be a function argument,
20    /// a local variable, a constant, or any type of data.
21    Value(LLVMValueRef), // https://llvm.org/doxygen/classllvm_1_1Value.html
22    /// Represents a basic block, which is a straight-line code sequence
23    /// with no branches in except to the entry and no branches out except at the exit.
24    BasicBlock(LLVMBasicBlockRef), // https://llvm.org/doxygen/classllvm_1_1BasicBlock.html
25    /// Provides a mechanism to construct an LLVM IR. Builders are the means
26    /// through which instructions are constructed for LLVM programs.
27    Builder(LLVMBuilderRef), // https://www.llvmpy.org/llvmpy-doc/0.12.7/doc/llvm.core.Builder.html
28    /// Represents a type in LLVM, which could be a primitive type (like integer or
29    /// floating-point), or a derived type (like arrays, pointers, structures, etc.).
30    Type(LLVMTypeRef), // https://llvm.org/doxygen/classllvm_1_1Type.html
31    /// Represents an LLVM execution engine, which is capable
32    /// of compiling and executing LLVM bitcode to native machine code.
33    ExecutionEngine(LLVMExecutionEngineRef) // https://llvm.org/doxygen/group__LLVMCExecutionEngine.html
34}

36/// Represents types of LLVM references for runtime conversion.
37#[derive(Debug, Clone, Copy)]
38pub enum LLVMRefType {
39    /// Refers to an LLVM context, which is an environment in which LLVM IR
40    /// generation and optimizations occur. All LLVM entities (modules, types, values)
41    /// belong to a specific context.
42    Context,
43    /// Represents an LLVM module, which acts as a container for function
44    /// definitions, global variables, and symbol declarations. 
45    Module,
46    /// Encompasses any value computed or used by a program. In LLVM, everything
47    /// from functions to variables to constants is a value.
48    Value,
49    /// A basic block is a sequence of instructions that execute sequentially.
50    /// It is a single-entry, single-exit section of code with no branches except at
51    /// the entry and the exit.
52    BasicBlock,
53    /// Used for building LLVM IR. It maintains a current position in a block
54    /// where new instructions are inserted.
55    Builder,
56    /// Represents a data type in LLVM. Types are metadata that describe the high-level
57    /// structure of data used within modules and provide the type system for LLVM IR.  
58    Type,
59    /// Manages the execution of compiled LLVM code. It is responsible for
60    /// compiling LLVM IR to machine code and executing or providing access to that code.
61    ExecutionEngine,
62}

64/// Helper methods for the LLVMRef enum to manage raw pointer conversions safely.
65impl LLVMRef {
66    /// Converts an LLVMRef to a raw pointer.
67    /// 
68    /// # Returns
69    /// A raw pointer to the underlying LLVM object.
70    fn to_raw(self) -> *mut c_void {
71        match self {
72            LLVMRef::Context(ptr) => ptr as *mut c_void,
73            LLVMRef::Module(ptr) => ptr as *mut c_void,
74            LLVMRef::Value(ptr) => ptr as *mut c_void,
75            LLVMRef::BasicBlock(ptr) => ptr as *mut c_void,
76            LLVMRef::Builder(ptr) => ptr as *mut c_void,
77            LLVMRef::Type(ptr) => ptr as *mut c_void,
78            LLVMRef::ExecutionEngine(ptr) => ptr as *mut c_void,
79        }
80    }
81
82    /// Constructs an LLVMRef from a raw pointer based on the specified kind.
83    /// 
84    /// # Parameters
85    /// * `ptr` - A raw pointer to the underlying LLVM object.
86    /// * `kind` - A kind of LLVMRef to create, which ensures the type safety of the raw pointer.
87    /// 
88    /// # Safety
89    /// This function is unsafe because it assumes the raw pointer is valid and properly typed.
90    /// 
91    /// # Returns
92    /// A new instance of LLVMRef corresponding to the specified type.
93    unsafe fn from_raw(ptr: *mut c_void, kind: LLVMRefType) -> Self {
94        match kind {
95            LLVMRefType::Context => LLVMRef::Context(ptr as LLVMContextRef),
96            LLVMRefType::Module => LLVMRef::Module(ptr as LLVMModuleRef),
97            LLVMRefType::Value => LLVMRef::Value(ptr as LLVMValueRef),
98            LLVMRefType::BasicBlock => LLVMRef::BasicBlock(ptr as LLVMBasicBlockRef),
99            LLVMRefType::Builder => LLVMRef::Builder(ptr as LLVMBuilderRef),
100            LLVMRefType::Type => LLVMRef::Type(ptr as LLVMTypeRef),
101            LLVMRefType::ExecutionEngine => LLVMRef::ExecutionEngine(ptr as LLVMExecutionEngineRef),
102        }
103    }
104}

106/// Thread-safe pointer type for managing raw C pointers in a synchronized context.
107pub struct SafeLLVMPointer {
108    ptr: Arc<RwLock<NonNull<c_void>>>,
109}
110
111impl SafeLLVMPointer {
112    /// Constructs a new `SafeLLVMPointer` by taking an `LLVMRef` and converting it to a non-null raw pointer.
113    /// 
114    /// # Parameters
115    /// * `llvm_ref` - An `LLVMRef` to be encapsulated.
116    /// 
117    /// # Returns
118    /// An `Option` wrapped `SafeLLVMPointer` if the pointer is non-null, `None` otherwise.
119    pub fn new(llvm_ref: LLVMRef) -> Option<Self> {
120        let raw_ptr = llvm_ref.to_raw();
121        NonNull::new(raw_ptr).map(|nn_ptr| SafeLLVMPointer {
122            ptr: Arc::new(RwLock::new(nn_ptr)),
123        })
124    }
125
126    /// Provides read-only access to the pointed-to value.
127    /// 
128    /// # Parameters
129    /// * `kind` - The type of LLVM reference expected.
130    /// * `f` - A closure that is executed with a reference to the value, allowing safe access.
131    /// 
132    /// # Type Parameters
133    /// * `Closure` - A closure or function pointer that takes a reference to an `LLVMRef`.
134    /// It performs operations based on the type of reference it receives.
135    /// * `ReturnType` - The return type of the closure `Closure`. This type is determined by
136    /// the return value of the closure. 
137    /// 
138    /// # Returns
139    /// The result of the closure, which can be any type defined by the closure's functionality.
140    /// 
141    /// # Errors
142    /// If the RwLock is poisoned, an error will be thrown, crashing the program or propagating the panic.
143    pub fn read<Closure, ReturnType>(&self, kind: LLVMRefType, closure: Closure) -> ReturnType
144    where
145        Closure: FnOnce(&LLVMRef) -> ReturnType,
146    {
147        let lock = self.ptr.read().expect("RwLock has been poisoned");
148        let ref_to_value = unsafe { LLVMRef::from_raw(lock.as_ptr(), kind) };
149        closure(&ref_to_value)
150    }
151
152    /// Provides write access to the pointed-to value.
153    /// 
154    /// # Parameters
155    /// * `kind` - The type of LLVM reference expected.
156    /// * `closure` - A closure that is executed with a mutable reference to the value, allowing safe modifications.
157    /// 
158    /// # Returns
159    /// The result of the closure, which can be any type defined by the closure's functionality.
160    /// 
161    /// # Errors
162    /// If the RwLock is poisoned, an error will be thrown, crashing the program or propagating the panic.
163    pub fn write<Closure, ReturnType>(&self, kind: LLVMRefType, closure: Closure) -> ReturnType
164    where
165        Closure: FnOnce(&mut LLVMRef) -> ReturnType,
166    {
167        let lock = self.ptr.write().expect("RwLock has been poisoned");
168        let mut ref_to_mut_value = unsafe { LLVMRef::from_raw(lock.as_ptr(), kind) };
169        closure(&mut ref_to_mut_value)
170    }
171}
