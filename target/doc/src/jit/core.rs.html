1//! The ExecutionEngine class manages the initialization and operation of an LLVM execution engine, along with its context and module.
2
3extern crate llvm_sys as llvm;
4use llvm::{core, execution_engine};
5use std::{any::Any, ffi::{c_char, CStr, CString}, sync::{Arc, RwLock}};
6use slog::Logger;
7use common::{pointer::{LLVMRef, LLVMRefType, SafeLLVMPointer}, target::{GeneralTargetConfigurator, TargetConfigurator}};
8
9/// Represents an LLVM execution engine for a multi-threaded environment.
10/// This struct encapsulates all necessary LLVM components: context, module, and execution engine.
11pub struct ExecutionEngine {
12    engine: Arc<RwLock<SafeLLVMPointer>>,
13    logger: Option<Logger>,
14}
15
16impl ExecutionEngine {
17    /// Constructs a new `ExecutionEngine`.
18    ///
19    /// This method initializes a new LLVM ExecutionEngine, configures the general target,
20    /// and optionally sets up a logger for debugging information based on the `debug_info` parameter.
21    ///
22    /// # Parameters
23    /// * `module` - A thread-safe `SafeLLVMPointer` containing an `LLVMModuleRef`.
24    /// * `debug_info` - If true, initializes a logger to record debugging information.
25    ///
26    /// # Returns
27    /// A new instance of `ExecutionEngine`.
28    pub fn new(module: Arc<RwLock<SafeLLVMPointer>>, debug_info: bool) -> Self {
29        GeneralTargetConfigurator.configure();
30        
31        let mut engine_ref: execution_engine::LLVMExecutionEngineRef = std::ptr::null_mut();
32        let mut out_error: *mut c_char = std::ptr::null_mut();
33        let engine_ptr = &mut engine_ref;
34
35        let module_rw_lock = module.try_read().expect("Failed to read module");
36        module_rw_lock.write(LLVMRefType::Module, |module_ref| {
37            if let LLVMRef::Module(module_ptr) = module_ref {
38                unsafe {
39                    if execution_engine::LLVMCreateExecutionEngineForModule(engine_ptr, *module_ptr, &mut out_error) != 0 {
40                        if !out_error.is_null() {
41                            let error_str = CStr::from_ptr(out_error).to_str().unwrap_or("Unknown error");
42                            eprintln!("{}", error_str);
43                            core::LLVMDisposeMessage(out_error);
44                            panic!("Failed to create execution engine");
45                        } else {
46                            panic!("Failed to create execution engine with unknown error.");
47                        }
48                    }
49                }
50            } else {
51                panic!("Module pointer is not correctly retrieved.");
52            }
53        });
54
55        let engine_cptr = SafeLLVMPointer::new(LLVMRef::ExecutionEngine(engine_ref)).expect("Engine cannot be null");
56
57        let logger = if debug_info {
58            Some(logging::core::init_logger())
59        } else {
60            None
61        };
62
63        Self {
64            engine: Arc::new(RwLock::new(engine_cptr)),
65            logger,
66        }
67    }
68
69    /// Configures the LLVM execution engine using a specified target configurator.
70    ///
71    /// # Parameters
72    /// * `target_configurator` - The target configurator to setup necessary LLVM targets.
73    ///
74    /// # Returns
75    /// Returns `Ok(())` on successful configuration and initialization, or `Err(String)` on failure,
76    /// including detailed error messages.
77    pub fn initialize_target<T: TargetConfigurator>(&mut self, target_configurator: T) -> Result<(), String> {
78        target_configurator.configure();
79        if let Some(logger) = &self.logger {
80            logging::core::log_info(&logger, "Target configured.");
81        }
82        Ok(())
83    }
84
85    /// Executes a specified function within the module.
86    ///
87    /// # Parameters
88    /// * `function_name` - The name of the function to be executed.
89    ///
90    /// # Returns
91    /// Returns `Ok(())` if the function is executed successfully, or `Err(String)` if an error occurs,
92    /// which could include the function not being found or an execution error.
93    /// Executes a specified function.
94    pub fn execute<ReturnType, ArgType>(&mut self, function_name: &str, args: ArgType) -> Result<ReturnType, String>
95    where
96        ReturnType: 'static, 
97        ArgType: Any + Send + Sync, 
98    {
99        let engine_lock = self.engine.try_read().map_err(|e| format!("Failed to obtain read lock on engine: {}", e))?;
100        let result = engine_lock.write(LLVMRefType::ExecutionEngine, |engine_ref| {
101            if let LLVMRef::ExecutionEngine(engine_ptr) = engine_ref {
102                let function_name_c = CString::new(function_name).map_err(|_| "Failed to create CString for function name.")?;
103                let function_address = unsafe { execution_engine::LLVMGetFunctionAddress(*engine_ptr, function_name_c.as_ptr()) };
104                
105                if function_address == 0 {
106                    if let Some(logger) = &self.logger {
107                        logging::core::log_warning(&logger, &format!("Function \"{}\" not found.", function_name));
108                    }
109                    return Err("Function not found in given module.".to_string());
110                }
111
112                unsafe {
113                    let func: extern "C" fn(ArgType) -> ReturnType = std::mem::transmute(function_address);
114                    Ok(func(args))
115                }
116            } else {
117                Err("Invalid engine pointer.".to_string())
118            }
119        });
120
121        match result {
122            Ok(value) => {
123                if let Some(logger) = &self.logger {
124                    logging::core::log_info(&logger, &format!("Function '{}' executed successfully.", function_name));
125                }
126                Ok(value)
127            },
128            Err(e) => {
129                if let Some(logger) = &self.logger {
130                    logging::core::log_error(&logger, &format!("Execution error: {}", e));
131                }
132                Err(e)
133            }
134        }
135    }
136}
